Conteo binario con botones

Componentes: 8 LEDs y 2 botones
Tutorial paso a paso:
1. Coloca 8 LEDs conectados a los pines 2 al 9.
2. A√±ade 2 botones: uno en pin 10 (suma) y otro en pin 11 (resta), con resistencias pull-down.
3. Escribe el siguiente c√≥digo en el editor:

void setup() {
  for (int i = 2; i <= 9; i++) {
    pinMode(i, OUTPUT);
  }
  pinMode(10, INPUT);
  pinMode(11, INPUT);
}

void loop() {
  if (digitalRead(10) == HIGH) {
    contador++;
    if (contador > 255) contador = 0;
    delay(300);
  }
  if (digitalRead(11) == HIGH) {
    contador--;
    if (contador < 0) contador = 255;
    delay(300);
  }

  for (int i = 0; i < 8; i++) {
    digitalWrite(2 + i, bitRead(contador, i));
  }
}
